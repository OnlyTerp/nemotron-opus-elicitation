```python
# Current: 8GB read √ó batch_size √ó epochs = impossible
# Fix: load once, index efficiently

class CSVDataset(Dataset):
    def __init__(self, csv_path):
        # Option 1: Memory-map (if fits in RAM)
        self.df = pd.read_csv(csv_path)  # Load ONCE
        
        # Option 2: If too large, use chunked/parquet
        # self.df = pd.read_parquet(csv_path.replace('.csv', '.parquet'))
        
    def __len__(self):
        return len(self.df)
    
    def __getitem__(self, idx):
        row = self.df.iloc[idx]  # O(1) index
        return torch.tensor(row.values), torch.tensor(row.target)
```
