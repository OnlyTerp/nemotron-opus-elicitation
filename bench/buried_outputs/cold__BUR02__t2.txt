// App.jsx
import { Suspense, lazy } from 'react';

const ProductPage = lazy(() => import('./pages/ProductPage'));

function App() {
  return (
    <Suspense fallback={<div>Loading product…</div>}>
      <Routes>
        <Route path="/product/:id" element={<ProductPage />} />
      </Routes>
    </Suspense>
  );
}

// pages/ProductPage.jsx
export default function ProductPage({ productId }) {
  // Fix the real bottleneck first: don't deep-clone props
  // Use memo, selectors, or immutable patterns instead
  const { name, price } = useSelector(state => ({
    name: state.products.byId[productId]?.name,
    price: state.products.byId[productId]?.price,
  }));

  return <div>{name} - ${price}</div>;
}
