```js
function debounce(fn, delay) {
  let timeoutId;
  return (...args) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn(...args), delay);
  };
}

const searchInput = document.getElementById('search');
const debouncedSearch = debounce(async (query) => {
  if (!query.trim()) return;
  const res = await fetch(`/api/search?q=${encodeURIComponent(query)}`);
  const data = await res.json();
  renderResults(data);
}, 300);

searchInput.addEventListener('input', (e) => debouncedSearch(e.target.value));
```

Key points: closure holds `timeoutId`, `clearTimeout` on every keystroke resets the timer, arrow function preserves `this` if needed. For aborting in-flight requests, wrap fetch in an AbortController and call `controller.abort()` inside the debounce before setting the new timeout.
